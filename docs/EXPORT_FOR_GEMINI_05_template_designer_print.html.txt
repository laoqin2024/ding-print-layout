{% block scripts %}
  <script>
    window.DESIGNER_PRINT_BOOT = {
      processCode: "{{ process_code }}",
      instanceId: "{{ instance_id }}",
      layoutName: "{{ layout_name }}",
      layout: {{ layout_json | safe }},
    };
  </script>
  <script>
    (function() {
      const boot = window.DESIGNER_PRINT_BOOT || {};
      const container = document.getElementById('pdf-container');
      const pageLabel = document.getElementById('page-label');
      const printFab = document.getElementById('print-fab');
      const backBtn = document.getElementById('back-btn');
      
      backBtn?.addEventListener('click', () => {
        window.history.back();
      });
      
      printFab?.addEventListener('click', () => {
        window.print();
      });
      
      // Generate PDF via designer preview API
      async function loadPDF() {
        try {
          const layout = boot.layout || {};
          
          const res = await fetch(`/designer/api/preview`, {
            method: 'POST',
            headers: { 'content-type': 'application/json' },
            body: JSON.stringify({
              process_code: boot.processCode,
              instance_id: boot.instanceId,
              base_pdf: layout.base_pdf || '',
              use_template: layout.use_template !== false,
              orientation: layout.orientation || 'l',
              cover_source_mode: layout.cover_source_mode || 'base',
              cover_mode: layout.cover_mode || 'strict',
              cover_offset_x: layout.cover_offset_x || 0,
              cover_offset_y: layout.cover_offset_y || 0,
              items: layout.items || [],
            }),
          });
          
          if (!res.ok) {
            const err = await res.json();
            throw new Error(err.msg || `HTTP ${res.status}`);
          }
          
          const blob = await res.blob();
          const url = URL.createObjectURL(blob);
          
          // Render PDF using embed
          if (container) {
            container.innerHTML = `
              <embed src="${url}" type="application/pdf" width="100%" height="800px" class="rounded-2xl" />
            `;
          }
          
          if (pageLabel) {
            pageLabel.textContent = 'PDF 已生成';
          }
        } catch (err) {
          console.error('PDF generation failed:', err);
          if (container) {
            container.innerHTML = `
              <div class="text-center py-14">
                <div class="glass rounded-3xl px-6 py-10 inline-block">
                  <div class="text-base font-black text-red-600 dark:text-red-400">PDF 生成失败</div>
                  <div class="text-[11px] text-slate-500 dark:text-slate-400 mt-2">${err.message || '未知错误'}</div>
                </div>
              </div>
            `;
          }
        }
      }
      
      loadPDF();
    })();
  </script>
{% endblock %}
